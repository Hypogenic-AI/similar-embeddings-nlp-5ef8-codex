% Bold paragraph headers (use: \para{Header.} followed by text)

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% RESULT HIGHLIGHTING
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% Colors for result arrows
\definecolor{pastelgreen}{RGB}{50,205,50}

\newcommand{\increase}{\textcolor{pastelgreen}{\bm{$\uparrow$}}}

\newcommand{\pstar}{\phantom{*}}
